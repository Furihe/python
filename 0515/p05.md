class Triangle:
    numberofSides = 3
    def __init__(self, a1, a2, a3):
        self.angle1 = a1
        self.angle2 = a2
        self.angle3 = a3
    def checkAngles(self):
        if self.angle1 + self.angle2 + self.angle3 == 180:
            return True
        else:
            return False
triangle = Triangle(90, 30, 60)
print(triangle.checkAngles())
