class Cat:
    def __init__(self,name,age):
        self.__name=name
        self.__age=age
    def __str__(self,):
        self.str_info=self.__name+" "+str(self.__age)
        return self.str_info
    def getAge(self,):
        return self.__age
    def getName(self,):
        return self.__name
    def setName(self,name):
        self.__name=name  
    def setAge(self,age):
        self.__age=age
missy=Cat('Missy',3)
lucky=Cat('Lucky',5)
print(missy)
print(lucky)
