import random

sLower="abcdefgfijklmnopqrstuvwxyz"
sNum="01234567890"
sUpper="ABCDEFGHIJKLMNOPQRSTUVWXYZ"
sSign="!@#$%^&*()?"
s=sLower+sNum+sUpper+sSign
passlen=8
sLCnt=0
sNCnt=0
sUCnt=0
sSCnt=0

while True:
    p="".join(random.sample(s,passlen))

   for i in p:
        if sLower.count(i)>0:
            sLCnt+=1
        elif sNum.count(i)>0:
            sNCnt+=1
        elif sUpper.count(i)>0:
            sUCnt+=1
        elif sSign.count(i)>0:
            sSCnt+=1

   if sLCnt==0 or sNCnt==0 or sUCnt==0 or sSCnt==0:
        print("강력한 암호는 소문자, 대문자, 숫자, 기호가 혼합되어야 합니다.")
        continue
    else:
        print(f"생성된 암호 = {p}")
        break
